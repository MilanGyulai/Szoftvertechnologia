Skeleton

Ebben a dokumentumban megtalálható lesz a játék Skeletonja, tehát az alapvető kódok a programhoz.

Inventory osztály

import java.util.ArrayList; import java.util.List; public class Inventory { private List items; private int capacity; public Inventory(int capacity) { this.capacity = capacity; this.items = new ArrayList<>(); } public boolean addItem(Item item) { if (items.size() < capacity) { items.add(item); return true; } return false; } public boolean removeItem(Item item) { return items.remove(item); } public List getItems() { return items; } }

Quest osztály

public class Quest { private String title; private String description; private boolean isCompleted; private List rewards; public Quest(String title, String description, List rewards) { this.title = title; this.description = description; this.isCompleted = false; this.rewards = rewards; } public void complete() { this.isCompleted = true; } public String getTitle() { return title; } public boolean isCompleted() { return isCompleted; } public List getRewards() { return rewards; } }

Crafting osztály

import java.util.Map; public class Crafting { private Map requiredMaterials; private Item craftedItem; public Crafting(Map requiredMaterials, Item craftedItem) { this.requiredMaterials = requiredMaterials; this.craftedItem = craftedItem; } public boolean canCraft(Inventory inventory) { for (Map.Entry material : requiredMaterials.entrySet()) { long count = inventory.getItems().stream() .filter(item -> item.getName().equals(material.getKey())) .count(); if (count < material.getValue()) { return false; } } return true; } public Item craftItem() { return craftedItem; } }
